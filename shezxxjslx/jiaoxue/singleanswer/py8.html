n=0 #人数
sa=0 #得A人数
sb=0
sc=0
sd=0
se=0
zf=0 #总分

ls=list(input().split())
while ls[0]!="-1":
    cj=eval(ls[1])*0.3+eval(ls[2])*0.7 #每个人的成绩

    if cj>=90:
        sa=sa+1
    elif cj>=80:
        sb=sb+1
    elif cj>=70:
        sc=sc+1
    elif cj>=60:
        sd=sd+1
    else:
        se=se+1
    n=n+1
    zf=zf+cj
    ls=list(input().split())

print(sa,sb,sc,sd,se,"\n""{:.2f}".format(zf/n))
